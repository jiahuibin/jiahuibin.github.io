公共js

var HHZ_COMMON = {
isDebug: true,
//获取域名
getHost: function () {
    var result;
    if (JD_COMMON.isDebug) {
        result = "http://test.huihaizhe.com/";//测试地址
    } else {
        result = "http://huihaizhe.com/";//正式地址

    return result;
},
//判断参数是否为null
isEmpty: function (text) {
    if (text == null || text == undefined || text == "undefined" || text === "") {
        return true;
    } else {
        return false;
    }
},
//判断参数不为null
isNotEmpty: function (text) {
    return !this.isEmpty(text);
},
//获取URL参数函数
getUrlParam: function (name) {
    var reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)");
    var r = window.location.search.substr(1).match(reg);
    if (r != null) {
        r = decodeURIComponent(unescape(r[2]));
        return this.stripscript(r);
    } else {
        return null;
    }
},
stripscript: function (s) {
    var pattern = new RegExp("[`~!@#$^&*()=|{}';',\\[\\].<>/?~！@#￥……&*（）—|{}【】‘；：”“'。，、？]");
    var rs = "";
    for (var i = 0; i < s.length; i++) {
        rs = rs + s.substr(i, 1).replace(pattern, '');
    }
    return rs;
},
//对ajax的封装
load: function (options) {
    //定义默认值
    this.defaultOpt = {
        type: 'GET',
        url: "",//必传
        data: {},//key-value
        jsonpCallback: 'callback',
        dataType: 'jsonp',//jsonp json
        msgTitle: "ajax 请求",//debug 时可以设置
        cache: false,//默认值true, dataType是'script'或'jsonp'时为false，可根据需要配置
        success: function (data) {
        },
        error: function (data) {
        },
    };
    var opt = $.extend(this.defaultOpt, options);
    /*对网络异常将来可以统一处理*/
    $.ajax({
        type: opt.type,
        url: opt.url,
        jsonpCallback: opt.jsonpCallback,
        dataType: opt.dataType,
        data: opt.data,
        error: function (data) {
            opt.error(data);
        },
        success: function (data) {
            opt.success(data);
        }
    });
},

常见问题

开发前可以避免

  • 杜绝为了完成工作而工作，拿到重构稿是认真分析页面结构。

    1. 提示框未展示问题

      产生该问题的原因是,提示框已经弹出，但是被页面上其它的遮罩层挡住。

      在获取到重构稿的时候，优先查看查看遮罩层的z-index属性值，如果未设置建议设置一个很大的值。

    2. 页面结构分析
      ①了解h5页面有多少给状态页面

      ②了解h5页面有多少个页面是全屏展示，多少个页面非全屏展示

    3. 可变高度和屏幕高度（可以解决大部分疑难问题）

开发进行中问题

　　1. h5页面唤起系统键盘切换不流畅问题

　　　产生该问题的原因是，iphone input和textarea 焦点失去时和系统键盘收起的响应时间问题。具体解决方法可以参考 ios键盘收起。（增加了延迟完美解决）

　　2. 冒泡问题

　　JQuery中阻止冒泡常用到的有以下3个方法：

①event.stopPropagation(); 只阻止了冒泡事件， 默认行为没有阻止

②event.preventDefault(); 只阻止了默认事件，冒泡事件没有阻止

③return false; 冒泡事件和默认事件都阻止
